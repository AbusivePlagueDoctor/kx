		icon = generic_political_discourse

			OR = {

					tooltip = COL_same_wars_as_Intl

				}
			}
		}

		visible = {
			has_socialist_government = yes
			is_in_faction = no

			NOT = { has_war_with = FROM }
			FROM = {
				has_socialist_government = yes
				is_faction_leader = yes
				has_capitulated = no
			}
		}

		targets = { INT }
		target_root_trigger = {
			has_completed_focus = COL_join_the_internationale
			is_in_faction = no
			is_subject = no
		}
		target_trigger = {
			NOT = { has_war_with = FROM }
			FROM = {
				has_socialist_government = yes
				is_faction_leader = yes
				has_capitulated = no
			}
		}

		fire_only_once = yes
		cost = 50

		ai_will_do = {
			factor = 10
			modifier = {
				factor = 0
				FROM = {
					is_faction_leader = yes
					has_socialist_government = yes
					surrender_progress > 0.30
				}
			}
		}
		complete_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision COL_join_international"
			INT = {
				country_event = { id = col.333 days = 1 }
			}
		}
	}
